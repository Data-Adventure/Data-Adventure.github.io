/*

 *@desc:dfs

 *@time:2022-12-31 10:24:41
 */

class Solution {
public:
    int res[31]{0, 1};
    int result{0};

    void dfs(int i, int n) {
        if (i > n) return;
        result = res[i - 1] + res[i - 2];
        res[i] = result;
        dfs(i + 1, n);
    }

    int fib(int n) {
        if (n == 0) return 0;
        else if (n == 1) return 1;
        dfs(2, n);
        return result;
    }
};

 *@number[509]:斐波那契数

class Solution {
public:
    int fib(int n) {
        if (n <= 1) return n;
        int dp[2];
        dp[0] = 0;
        dp[1] = 1;
        for (int i = 2; i <= n; i ++) {
            int sum = dp[0] + dp[1];
            dp[0] = dp[1];
            dp[1] = sum;
        }
        return dp[1];
    }
};

 *@desc:
 *@reference:
 *@time:2022-01-03 09:59:32
 */

class Solution {
public:
    int minCostClimbingStairs(vector<int> &cost) {
        int pre{0}, cur{0};
        for (int i = 2; i < cost.size() + 1; i ++) {
            int result = min(cur + cost[i - 1], pre + cost[i - 2]);
            pre = cur;
            cur = result;
        }
        return cur;
    }
};

[62]:不同路径

/*
 *@number[62]:不同路径
 *@author:lls86
 *@desc:每个点代表从start到此点的方法
 *@reference:
 *@time:2023-01-03 20:05:52
 */

class Solution {
public:

    int uniquePaths(int m, int n) {
        int map[m][n];
        // 0列初始化
        for (int i = 0; i < m; i ++) {
            map[i][0] = 1;
        }
        // 0行初始化
        for (int i = 0; i < n; i ++) {
            map[0][i] = 1;
        }
        for (int i = 1; i < m; i ++) {
            for (int j = 1; j < n; j ++) {
                map[i][j] = map[i - 1][j] + map[i][j - 1];
            }
        }
        return map[m - 1][n - 1];
    }
};

[63]:不同路径 II

/*
 *@number[63]:不同路径 II
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-03 20:59:36
 */

class Solution {
public:
    int uniquePathsWithObstacles(vector<vector<int>> &obstacleGrid) {
        int m = obstacleGrid.size();
        int n = obstacleGrid[0].size();
        if (obstacleGrid[m - 1][n - 1] == 1 || obstacleGrid[0][0] == 1) return 0;
        vector<vector<int>> dp(m, vector<int>(n, 0));
        for (int i = 0; i < m && obstacleGrid[i][0] == 0; i ++) {
            dp[i][0] = 1;
        }
        for (int j = 0; j < n && obstacleGrid[0][j] == 0; j ++) {
            dp[0][j] = 1;
        }
        for (int i = 1; i < m; i ++) {
            for (int j = 1; j < n; j ++) {
                if (obstacleGrid[i][j] == 1) continue;
                dp[i][j] = dp[i - 1][j] + dp[i][j - 1];
            }
        }
        return dp[m - 1][n - 1];
    }
};

/*
 *@number[343]:整数拆分
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-04 10:23:42
 */

class Solution {
public:
    int integerBreak(int n) {
        vector<int> dp(n + 1, 1);
        for (int i = 3; i <= n; i ++) {
            for (int j = 1; j <= i / 2; j ++) {
                dp[i] = max(dp[i], max((i - j) * j, dp[i - j] * j));
            }
        }
        return dp[n];
    }
};

class Solution {
public:
    int numTrees(int n) {
        vector<int> dp(n + 1);
        dp[0] = 1;
        for (int i = 1; i <= n; i ++) {
            for (int j = 1; j <= i; j ++) {
                dp[i] += dp[j - 1] * dp[i - j];
            }
        }
        return dp[n];
    }
};

[416]:分割等和子集

/*
 *@number[416]:分割等和子集
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-05 15:22:10
 */

class Solution {
public:
    bool canPartition(vector<int> &nums) {
        sort(nums.begin(), nums.end());
        int sum{0};
        for (int i = 0; i < nums.size(); i ++) {
            sum += nums[i];
        }
        // 奇数的时候返回false
        if (sum % 2 != 0) return false;
        else {
            int target = sum / 2;
            int dp[10001]{0};  // dp[i]取nums[i]或不取nums[i]的总和
            // 滚动数组
            for (int i = 0; i < nums.size(); i ++) {
                for (int j = target; j >= nums[i]; j --) { // 每一个元素一定是不可重复放入，所以从大到小遍历
                    dp[j] = max(dp[j], dp[j - nums[i]] + nums[i]);
                }
            }
            // 集合中的元素正好可以凑成总和target
            if (dp[target] == target) return true;
            return false;
        }
    }
};

[1049]:最后一块石头的重量 II

/*
 *@number[1049]:最后一块石头的重量 II
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-05 16:05:08
 */

class Solution {
public:
    int lastStoneWeightII(vector<int> &stones) {
        vector<int> dp(3001, 0);
        int sum = 0;
        for (int i = 0; i < stones.size(); i ++) sum += stones[i];
        int target = sum / 2;
        for (int i = 0; i < stones.size(); i ++) { // 遍历物品
            for (int j = target; j >= stones[i]; j --) { // 遍历背包
                dp[j] = max(dp[j], dp[j - stones[i]] + stones[i]);
            }
        }
        return sum - dp[target] - dp[target];
    }
};

[494]:目标和

/*
 *@number[494]:目标和
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-05 16:52:44
 */

class Solution {
public:
    int findTargetSumWays(vector<int>& nums, int target) {
        int sum = 0;
        for (int i = 0; i < nums.size(); i++) sum += nums[i];
        if (abs(target) > sum) return 0; // 此时没有方案
        if ((target + sum) % 2 == 1) return 0; // 此时没有方案
        int bagSize = (target + sum) / 2;
        vector<int> dp(bagSize + 1, 0); // dp[j]表示:填满j（包括j）这么大容积的包，有dp[j]种方法
        dp[0] = 1;
        for (int i = 0; i < nums.size(); i ++) {
            for (int j = bagSize; j >= nums[i]; j --) {
                dp[j] += dp[j - nums[i]];
            }
        }
        return dp[bagSize];
    }
};

class Solution {
public:
    int change(int amount, vector<int> &coins) {
        // dp[j]:凑成总金额j的货币组合数为dp[j]
        vector<int> dp(amount + 1, 0);
        dp[0] = 1;
        for (int i = 0; i < coins.size(); i ++) {
            for (int j = coins[i]; j <= amount; j ++) {
                dp[j] += dp[j - coins[i]];
            }
        }
        return dp[amount];
    }

        dp[0] = 1;
        for (int i = 0; i <= target; i ++) {
            for (int j = 0; j < nums.size(); j ++) {

        dp[0] = 1;
        for (int i = 1; i <= n; i ++) {

/*
 *@number[322]:零钱兑换
 *@author:lls86
 *@desc:dp
 *@reference:
 *@time:2023-01-13 16:51:01
 */

class Solution {
public:
    int coinChange(vector<int> &coins, int amount) {
        // dp[i]:满足条件的硬币最少数
        vector<int> dp(amount + 1, INT_MAX);
        dp[0] = 0;
        for (int i = 0; i < coins.size(); i ++) {
            for (int j = coins[i]; j <= amount; j ++) {
                if (dp[j - coins[i]] != INT_MAX) {
                    dp[j] = min(dp[j - coins[i]] + 1, dp[j]);
                }
            }
        }
        if (dp[amount] == INT_MAX) return - 1;
        return dp[amount];
    }
};

class Solution {
public:
    int numSquares(int n) {
        vector<int> dp(n + 1, INT_MAX);
        dp[0] = 0;
        for (int i = 0; i <= n; i ++) {
            for (int j = 1; j * j <= i; j ++) {
                dp[i] = min(dp[i - j * j] + 1, dp[i]);
            }
        }

    }
};

/*
 *@number[139]:单词拆分
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-13 17:24:20
 */

class Solution {
public:
    bool wordBreak(string s, vector<string> &wordDict) {
        unordered_set<string> wordSet(wordDict.begin(), wordDict.end());
        vector<bool> dp(s.size() + 1, false);
        dp[0] = true;
        for (int i = 1; i <= s.size(); i ++) {   // 遍历背包
            for (int j = 0; j < i; j ++) {       // 遍历物品
                string word = s.substr(j, i - j); //substr(起始位置，截取的个数)
                if (wordSet.find(word) != wordSet.end() && dp[j]) {
                    dp[i] = true;
                }
            }
        }
        return dp[s.size()];
    }
};

/*
 *@number[198]:打家劫舍
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-13 17:48:59
 */

class Solution {
public:
    int rob(vector<int> &nums) {
        if (! nums.size()) return 0;
        if (nums.size() == 1) return nums[0];
        vector<int> dp(nums.size(), 0);
        dp[0] = nums[0];
        dp[1] = max(nums[0], nums[1]);
        for (int i = 2; i < nums.size(); i ++) {
            dp[i] = max(dp[i - 2] + nums[i], dp[i - 1]);
        }
        return dp[nums.size() - 1];
    }
};

[213]:打家劫舍 II

/*
 *@number[213]:打家劫舍 II
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-13 18:07:42
 */

class Solution {
public:
    int rob(vector<int> &nums) {
        if (nums.size() == 0) return 0;
        if (nums.size() == 1) return nums[0];
        int result1 = robRange(nums, 0, nums.size() - 2); // 情况二
        int result2 = robRange(nums, 1, nums.size() - 1); // 情况三
        return max(result1, result2);
    }
    
    int robRange(vector<int> &nums, int start, int end) {
        if (end == start) return nums[start];
        vector<int> dp(nums.size());
        dp[start] = nums[start];
        dp[start + 1] = max(nums[start], nums[start + 1]);
        for (int i = start + 2; i <= end; i ++) {
            dp[i] = max(dp[i - 2] + nums[i], dp[i - 1]);
        }
        return dp[end];
    }
};

[337]:打家劫舍 III

/*
 *@number[337]:打家劫舍 III
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-14 14:55:51
 */

class Solution {
public:
    vector<int> robtree(TreeNode *cur) {
      	// 后序遍历
        if (! cur) return vector<int>{0, 0};
        vector<int> left = robtree(cur->left);
        vector<int> right = robtree(cur->right);

        // 偷cur，则无左右节点
        int val1 = cur->val + left[0] + right[0];
        // 不偷，则有左右节点
        int val2 = max(left[0], left[1]) + max(right[0], right[1]);
        return {val2, val1};
    }

    int rob(TreeNode *root) {
        vector<int> result = robtree(root);
        return max(result[0], result[1]);
    }
};

[121]:买卖股票的最佳时机

/*
 *@number[121]:买卖股票的最佳时机
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-14 15:46:57
 */

class Solution {
public:
    int maxProfit(vector<int> &prices) {
        // dp[i][0 1]:第i天持有股票，不持有股票
        int len = prices.size();
        if (! len) return 0;
        vector<vector<int>> dp(2, vector<int>(2));
        dp[0][0] = - prices[0];
        dp[0][1] = 0;
        for (int i = 1; i < len; i ++) {
            dp[i % 2][0] = max(dp[(i - 1) % 2][0], - prices[i]);
            dp[i % 2][1] = max(dp[(i - 1) % 2][1], dp[(i - 1) % 2][0] + prices[i]);
        }
        return dp[(len - 1) % 2][1];
    }
};

[122]:买卖股票的最佳时机 II

/*
 *@number[122]:买卖股票的最佳时机 II
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2023-01-14 16:44:53

class Solution {
public:

    int maxProfit(vector<int> &prices) {
        int len = prices.size();
        vector<vector<int>> dp(2, vector<int>(2));
        dp[0][0] -= prices[0];
        dp[0][1] = 0;
        for (int i = 1; i < len; i ++) {
            dp[i % 2][0] = max(dp[(i - 1) % 2][0], dp[(i - 1) % 2][1] - prices[i]);
            dp[i % 2][1] = max(dp[(i - 1) % 2][1], prices[i] + dp[(i - 1) % 2][0]);
        }
        return dp[(len - 1) % 2][1];
    }
};
