#include <iostream>

typedef long long ll;
using namespace std;

int main(){
    ll n, t, d, b;
    cin >> n >> t;
    
    ll ans{0}, tmp{0}; // tmp 暂时能吃的天数
    while (n --) {
        cin >> d >> b;
        if  (tmp >= d) d = tmp + 1;
        tmp = d + b - 1;
        ans += b;
        if (tmp >= t){
            ans -= tmp - t;
            break;
        }
    }
    cout << ans << endl;
    return 0;
}

// 4440. 照相
#include <iostream>

// 类似翻硬币题目，无论如何均可全部翻转成0（HG），但是要求最少次数
// 下界是相邻w[]不同的次数，当最后是1的时候，需要将整体翻转，0时，无需翻转
// w[]全部为0，无需特判
using namespace std;

const int N = 2e5 + 10;

int n, m;
char str[N];
int w[N];

int main(){
    scanf("%d", &n);
    scanf("%s", str);
    for(int i = 0; i < n; i += 2){
        if  (str[i] != str[i + 1]){
            w[m ++] = str[i] == 'G';
        }
    }
    int res = 0;
    for (int i = 0; i < m; i ++){
        if  (w[i] != w[i + 1]){
            res ++;    
        }
        
    }
    printf("%d\n", res);
}

#include<iostream>

using namespace std;
typedef long long ll;

int main() {
    int T;
    scanf("%d", &T);
    for (int i = 1; i <= T; i++) {
        int L, n, a, sum = 0;
        ll cnt = 0;
        char op;
        scanf("%d%d", &L, &n);
        while (n--) {
            scanf("%d %c", &a, &op);
            if (op == 'C') sum += a;
            else sum -= a;
            cnt += abs(sum) / L;
            sum %= L;
        }
        printf("Case #%d: %lld\n", i, cnt);
    }
    return 0;
}

// 4742. 电

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 2e5 + 10, M = 2 * N;

int n;
int h[N], e[M], ne[M], idx;
int w[N];
int f[N];

void add(int a, int b){
    e[idx] = b, ne[idx] = h[a], h[a] = idx ++;
}

int dp(int u){
    if (f[u] != -1) return f[u];
    int res = 1;
    for (int i = h[u]; ~i; i = ne[i]){
        int j = e[i];
        if (w[u] > w[j]) res += dp(j);
    }
    f[u] = res;
    return res;
}

int main()
{
    int T;
    scanf("%d", &T);
    for(int cases = 1; cases <= T; cases ++){
        scanf("%d", &n);
        for (int i = 1; i <= n; i ++) scanf("%d", &w[i]);
        memset(h, -1, (n + 1) * 4);
        idx = 0;
        for(int i = 0; i < n - 1; i ++){
            int a, b;
            scanf("%d%d", &a, &b);
            add(a, b), add(b, a);
        }
        memset(f, -1, (n + 1) * 4);
        int res = 0;
        for(int i = 1; i <= n; i ++){
            res = max(res, dp(i));
        }
        printf("Case #%d: %d\n", cases, res);
    }
    return 0;
}

// 4736 步行者
#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int T, m, n, p;
int path[2][32];

int main() {
    cin >> T;
    for (int t = 1; t <= T; t++) {
        cin >> m >> n >> p;
        for (int i = 1; i <= m; i++) {
            int route = 0;
            for (int j = 1; j <= n; j++) {
                scanf("%d", &route);
                if (i == p) {
                    path[1][j] = route;
                }
                path[0][j] = max(route, path[0][j]);
            }
        }
        long long res = 0;
        for (int i = 1; i <= n; i++) {
            if (path[1][i] < path[0][i]) {
                res += (path[0][i] - path[1][i]);
            }
        }
        printf("Case #%d: %lld\n", t, res);
        memset(path, 0, sizeof path); // 重置数组
    }
    return 0;
}

// 4737. 冰壶
#include <iostream>
#include <algorithm>

using namespace std;

const int N = 10;

int T, rs, rh, n, m;
double da[N], db[N];

int main() {
    scanf("%d", &T);

    int x, y, r;
    for (int k = 1; k <= T; k++) {
        scanf("%d%d", &rs, &rh);
        r = da[0] = db[0] = (rs + rh) * (rs + rh);

        scanf("%d", &n);
        for (int i = 0; i < n; i++) {
            scanf("%d%d", &x, &y);
            da[i] = x * x + y * y;
        }
        scanf("%d", &m);
        for (int i = 0; i < m; i++) {
            scanf("%d%d", &x, &y);
            db[i] = x * x + y * y;
        }
        sort(da, da + n), sort(db, db + m);

        int a = 0, b = 0;
        while (a < n && da[a] <= db[0] && da[a] <= r) a++;
        while (b < m && db[b] <= da[0] && db[b] <= r) b++;

        printf("Case #%d: %d %d\n", k, a, b);
    }
    return 0;
}

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

const int N = 1010;
int tt, n, Case;

//q1 按照 cmp1 的方式排序
//q2 按照 cmp2 的方式排序
struct node {
    string c;
    int d, u;
} q1[N], q2[N];

bool cmp1(node a, node b) {
    return a.c < b.c || a.c == b.c && a.u < b.u;
}

bool cmp2(node a, node b) {
    return a.d < b.d || a.d == b.d && a.u < b.u;
}

int main() {
    scanf("%d", &tt);
    while (tt--) {
        scanf("%d", &n);
        for (int i = 0; i < n; i++) {
            cin >> q1[i].c;
            scanf("%d%d", &q1[i].d, &q1[i].u);
            //要将 q1 的所有数据都 copy 进 q2
            q2[i] = q1[i];
        }
        //分别排好序
        sort(q1, q1 + n, cmp1);
        sort(q2, q2 + n, cmp2);
        int res = 0;
        //判断有几个位置相等并且值也相等
        for (int i = 0; i < n; i++)
            if (q1[i].c == q2[i].c && q1[i].d == q2[i].d && q1[i].u == q2[i].u)
                res++;
        printf("Case #%d: %d\n", ++Case, res);
    }
    return 0;
}

#include <iostream>
#include <cstring>
#include <algorithm>

using namespace std;

int T;

int main()
{
    scanf("%d", &T);
    
    for(int t = 1; t <= T; t ++){
        string s, f;
        cin >> s >> f;
        int res = 0;//结果
        for (int i = 0; i < s.size(); i++)
        {
            int ans = 26;//记录每个最小差值(差值必定小于26)
            for (int j = 0; j < f.size(); j++)
            {
                //从三者中取最小值
                ans = min(ans, abs(s[i] - f[j]));
                ans = min(ans, abs(s[i] - 26 - f[j]));
                ans = min(ans, abs(s[i] + 26 - f[j]));
            }
            res += ans;//求和
        }
        printf("Case #%d: %d\n", t, res);
    }
    return 0;
}

#include <iostream>
#include <cstring>
#include <algorithm>
#include <vector>

using namespace std;

const int N = 1e5 + 10;

int T, n, x;
struct position {
    int out, number;
} p[N];


bool cmp(position a, position b) {
    if (a.out != b.out) {
        return a.out < b.out;
    } else {
        return a.number < b.number;
    }
}

int main() {
    scanf("%d", &T);

    for (int i = 1; i <= T; i++) {
        scanf("%d%d", &n, &x);
        int a = 0;
        for (int j = 1; j <= n; j++) {
            scanf("%d", &a);
            a = (a - 1) / x;
            p[j].out = a, p[j].number = j;
        }
        sort(p + 1, p + n + 1, cmp);
        
        printf("Case #%d: ", i);
        for (int t = 1; t <= n; t++) {
            printf("%d ", p[t].number);
        }
        printf("\n");
    }

    return 0;
}

#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;
const int N = 1e5 + 10;
int n;
int q[N], w[N];

int main() {

    int T;
    scanf("%d", &T);
    for (int t = 1; t <= T; t++) {

        scanf("%d", &n);
        for (int i = 0; i < n; i++) {
            scanf("%d", &q[i]);
        }
        memcpy(w, q, sizeof w);
        sort(w, w + n);
        printf("Case #%d: ", t);
        for (int i = 0; i < n; ++i) {
            int l = 0, r = n - 1;
            while (l < r) {
                int mid = l + r + 1 >> 1;
                if (w[mid] <= 2 * q[i]) l = mid;
                else r = mid - 1;
            }
            if (w[r] == q[i]) r--;
            if (r < 0) printf("-1 ");
            else printf("%d ", w[r]);
        }
        puts("");
    }
    return 0;
}

#include<iostream>

using namespace std;
#define PI 3.1415926

int main() {
    int T;
    cin >> T;
    for (int t = 1; t <= T; t++) {
        int r, a, b;
        cin >> r >> a >> b;

        double res = 0;
        bool tag = true;//tag==true表示下一个是左圆，false则表示下一个是右圆
        while (r != 0) {
            res += (double) PI * r * r;
            if (tag) {
                r *= a;
                tag = false;
            } else {
                r /= b;
                tag = true;
            }
        }
        printf("Case #%d: %.6lf\n", t, res);
    }
    return 0;
}

#include <iostream>
#include <string>

using namespace std;

int T, N, D, M;
long long C;

bool judge(string s, int i) {
    while (i < s.length())
        if (s[i++] == 'D')
            return false;
    return true;
}

int main() {
    scanf("%d", &T);

    for (int t = 1; t <= T; t++) {
        scanf("%d%d%lld%d", &N, &D, &C, &M);
        string s;
        cin >> s;

        int i = 0;
        for (; i < s.size(); i++) {
            if (s[i] == 'C') {
                if (C == 0) {
                    break;
                } else {
                    C--;
                }
            } else if (s[i] == 'D') {
                if (D == 0) {
                    break;
                } else {
                    D--;
                    C += M;
                }
            }
        }
        printf("Case #%d: %s\n", t, judge(s, i) ? "YES" : "NO");
    }
    return 0;
}
