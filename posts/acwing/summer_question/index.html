// 4908. 饥饿的牛

#include <iostream>

typedef long long ll;
using namespace std;

int main(){
    ll n, t, d, b;
    cin >> n >> t;
    
    ll ans{0}, tmp{0}; // tmp 暂时能吃的天数
    while (n --) {
        cin >> d >> b;
        if  (tmp >= d) d = tmp + 1;
        tmp = d + b - 1;
        ans += b;
        if (tmp >= t){
            ans -= tmp - t;
            break;
        }
    }
    cout << ans << endl;
    return 0;
}
