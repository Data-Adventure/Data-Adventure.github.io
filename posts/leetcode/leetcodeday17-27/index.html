/*
 *@number[110]:平衡二叉树
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-03 16:08:46
 */
class Solution {
public:
    int height(TreeNode *node) {
        if (! node) return 0;
        else {
            return max(height(node->left), height(node->right)) + 1;
        }
    }

    bool isBalanced(TreeNode *root) {
        if (! root) return true;
        else {
            return abs(height(root->left) - height(root->right)) <= 1 && isBalanced(root->left) &&
                   isBalanced(root->right);
        }
    }
};
  • 自底向上：后续遍历，没有重复计算

对于当前遍历到的节点，先递归地判断其左右子树是否平衡，再判断以当前节点为根的子树是否平衡。如果一棵子树是平衡的，则返回其高度（高度一定是非负整数），否则返回 -1−1。如果存在一棵子树不平衡，则整个二叉树一定不平衡。

/*
 *@number[110]:平衡二叉树
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-03 16:08:46
 */
class Solution {
public:
    int height(TreeNode *node) {
        if (! node) return 0;
        int lefthight = height(node->left); // 左子树高度
        int rightheight = height(node->right);  // 右子树高度
        // 遍历之后进行判断
        if (lefthight == - 1 || rightheight == - 1 || abs(lefthight - rightheight) > 1) {
            return - 1;
        } else {
            return max(lefthight, rightheight) + 1;
        }
    }

    bool isBalanced(TreeNode *root) {
        return height(root) >= 0;
    }
};

/*
 *@number[257]:二叉树的所有路径
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-04 15:08:45
 */

class Solution {
private:
    void traversal(TreeNode *cur, vector<int> &path, vector<string> &result) {
        path.push_back(cur->val);  // 前序遍历，在遍历左右子树之前，加入path
        if (cur->left == nullptr && cur->right == nullptr) {
            string spath;  // 如果到了叶子节点
            for (int i = 0; i < path.size() - 1; i ++) {
                spath += to_string(path[i]);
                spath += "->";
            }
            spath += to_string(path[path.size() - 1]);
            result.push_back(spath);
            return;
        }
        if (cur->left) {
            traversal(cur->left, path, result);
            path.pop_back();  // 回溯
        }
        if (cur->right) {
            traversal(cur->right, path, result);
            path.pop_back();
        }
    }

public:
    vector<string> binaryTreePaths(TreeNode *root) {
        vector<string> result;
        vector<int> path;
        if (! root) return result;
        traversal(root, path, result);
        return result;
    }
};

/*
 *@number[404]:左叶子之和
 *@author:lls86
 *@desc:递归，后序遍历
 *@reference:
 *@time:2022-12-04 19:36:04
 */

class Solution {
public:
    int sumOfLeftLeaves(TreeNode *root) {
        // 后序遍历（左右中），通过递归函数的返回值来累加求取左叶子数值之和
        if (! root) return 0;
        if (! root->left && ! root->right) return 0;
        int leftvalue = sumOfLeftLeaves(root->left);
        // 必须要通过节点的父节点来判断其左孩子是不是左叶子
        if (root->left && ! root->left->left && ! root->left->right) {
            leftvalue = root->left->val;
        }
        int rightvalue = sumOfLeftLeaves(root->right);
        int res = leftvalue + rightvalue;
        return res;
    }
};

/*
 *@number[404]:左叶子之和
 *@author:lls86
 *@desc:迭代但是使用队列
 *@reference:
 *@time:2022-12-04 19:36:04
 */
class Solution {
public:
    int sumOfLeftLeaves(TreeNode *root) {
        if (! root->left && ! root->right) return 0;
        int res{};
        queue<TreeNode *> que;
        que.emplace(root);
        while (! que.empty()) {
            int size = que.size();
            for (int i = 0; i < size; i ++) {
                TreeNode *node = que.front();
                que.pop();
                if (node->left != nullptr && node->left->left == nullptr && node->left->right == nullptr) {
                    res += node->left->val;
                }
                if (node->right) que.push(node->right);
                if (node->left) que.push(node->left);
            }
        }
        return res;
    }
};

 *@author:lls86
 *@desc:层遍历，取每层第一个
 *@reference:
 *@time:2022-12-04 20:39:02

class Solution {
public:
    int findBottomLeftValue(TreeNode *root) {

        int res;
        queue<TreeNode *> que;
        que.push(root);
        while (! que.empty()) {
            int size = que.size();
            for (int i = 0; i < size; i ++) {
                TreeNode *node = que.front();
                que.pop();
                if (i == 0 && ! node->left && ! node->left) res = node->val;
                if (node->left) que.emplace(node->left);
                if (node->right) que.emplace(node->right);
            }
        }
        return res;
    }
};

/*
 *@number[513]:找树左下角的值
 *@author:lls86
 *@desc:dfs,更快
 *@reference:
 *@time:2022-12-04 20:39:02
 */
class Solution {
public:
    void dfs(TreeNode *node, int height, int &curval, int &curheight) {
        if (! node) return;
        height ++;
      	// 后序遍历，先遍历左子树
        dfs(node->left, height, curval, curheight);
        dfs(node->right, height, curval, curheight);
        // curval 记录高度在curheight的最左节点的值。在深度优先搜索时，我们先搜索当前节点的左子节点，再搜索当前节点的右子节点，然后判断当前节点的高度height是否大于curheight，如果是，那么将curval设置为当前结点的值，curheight设置为height
        if (height > curheight) {
            curheight = height;
            curval = node->val;
        }
    }

    int findBottomLeftValue(TreeNode *root) {
        int curval{}, curheight{};
        dfs(root, 0, curval, curheight);
        return curval;
    }
};

class Solution {
public:
    bool hasPathSum(TreeNode *root, int targetSum) {
        if (! root) return false;
        // node,int 节点与值
        queue<pair<TreeNode *, int>> que;
        que.emplace(root, root->val);
        while (! que.empty()) {
            int size = que.size();
            // cout << "que.size() " << que.size() << "\n";
            for (int i = 0; i < size; i ++) {
                // cout << "i " << i << " " << que.front().second << "\n";
                pair<TreeNode *, int> node = que.front();
                que.pop();
                // 妙处，如果叶节点有目标值提前跳出，且可以过滤掉非目标节点
                if (!node.first->left && !node.first->right && targetSum == node.second) return true;
                if (node.first->right) que.emplace(node.first->right, node.second + node.first->right->val);
                if (node.first->left) que.emplace(node.first->left, node.second + node.first->left->val);
            }
        }
        return false;
    }
};

/*
 *@number[112]:路径总和
 *@author:lls86
 *@desc:dfs
 *@reference:
 *@time:2022-12-04 21:20:39
 */
class Solution {
public:
    bool hasPathSum(TreeNode *root, int sum) {
        if (root == nullptr) {
            return false;
        }
        // 如果是叶子节点，且值正确，返回true
        if (root->left == nullptr && root->right == nullptr) {
            return sum == root->val;
        }
        // 否则不是叶子节点，则计算对应的值并寻找其叶子节点。 sum - root->val 值的变化
        return hasPathSum(root->left, sum - root->val) ||
               hasPathSum(root->right, sum - root->val);
    }
};

class Solution {
public:
    vector<vector<int>> ret;
    vector<int> path;

    void dfs(TreeNode *root, int targetSum) {
        if (! root) return;
        path.emplace_back(root->val);
        // for (int i = 0; i < path.size(); i ++) {
        //     cout << path[i] << " ";
        // }
        targetSum -= root->val;
        if (! root->left && ! root->right && targetSum == 0) {
            ret.emplace_back(path);
        }
        dfs(root->left, targetSum);
        dfs(root->right, targetSum);
        path.pop_back(); // 弹出最后一个元素
        // 结束位置，返回栈的上一层
    }

    vector<vector<int>> pathSum(TreeNode *root, int targetSum) {
        dfs(root, targetSum);
        return ret;
    }
};

/*
 *@number[106]:从中序与后序遍历序列构造二叉树
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-05 14:02:48

#include "../func.h"

//leetcode submit region begin(Prohibit modification and deletion)
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    unordered_map<int, int> pos;

    TreeNode *buildTree(vector<int> &inorder, vector<int> &postorder) {
        int n = inorder.size();
        for (int i = 0; i < n; i ++) {
            pos[inorder[i]] = i;     //记录中序遍历的根节点位置
        }
        return dfs(inorder, postorder, 0, n - 1, 0, n - 1);
    }

    TreeNode *dfs(vector<int> &inorder, vector<int> &postorder, int inorder_left, int inorder_right, int postorder_left,
                  int postorder_right) {
        if (inorder_left > inorder_right) return nullptr;
        int k = pos[postorder[postorder_right]];   //中序遍历根节点位置
        TreeNode *root = new TreeNode(postorder[postorder_right]); //创建根节点
        root->left = dfs(inorder, postorder, inorder_left, k - 1, postorder_left,
                         postorder_left + k - 1 - inorder_left);
        root->right = dfs(inorder, postorder, k + 1, inorder_right, postorder_left + k - 1 - inorder_left + 1,
                          postorder_right - 1);
        return root;
    }
};

/*
 *@number[105]:从前序与中序遍历序列构造二叉树
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-05 14:45:10
 */
s Solution {
public:

    // 记录每个节点在中序遍历的[值,位置]
    unordered_map<int, int> pos;

    TreeNode *dfs(vector<int> &pre, vector<int> &in, int pre_left, int pre_right, int in_left, int in_right) {
        if (pre_left > pre_right) return nullptr;
        // k 表示 前序遍历 根结点在 中序遍历 范围区间内的位置（分割左右子树）
        int k = pos[pre[pre_left]] - in_left;

        // 使用前序遍历根节点的位置去获取中序遍历位置
        TreeNode *root = new TreeNode(pre[pre_left]);
        // 左子树的中序遍历序列(in_left ~ in_left + k - 1),右子树的中序遍历序列(in_left + k + 1 ~ in_right)
        // 左子树的前序遍历序列(pre_left + 1 ~ pre_left + k),右子树的前序遍历序列(pre_left + k + 1, pre_right)
        // cout << "root->left:" << k << endl;
        // cout << pre_left + 1 << " " << pre_left + k << " " << in_left << " " << in_left + k - 1 << " ";
        // cout << "\n";
        root->left = dfs(pre, in, pre_left + 1, pre_left + k, in_left, in_left + k - 1);
        // cout << "root->right:" << k << endl;
        // cout << pre_left + k + 1 << " " << pre_right << " " << in_left + k + 1 << " " << in_left + k + 1 << " ";
        // cout << "\n";
        root->right = dfs(pre, in, pre_left + k + 1, pre_right, in_left + k + 1, in_right);
        return root;
    }

    TreeNode *buildTree(vector<int> &preorder, vector<int> &inorder) {
        int n = preorder.size();
        // 记录每个结点在中序遍历序列中的位置
        for (int i = 0; i < n; i ++) {
            pos[inorder[i]] = i;
        }
        return dfs(preorder, inorder, 0, n - 1, 0, n - 1);
    }
};

/*
 *@number[654]:最大二叉树
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-05 16:00:38
 */

class Solution {
public:
    TreeNode *construct(const vector<int> &nums, int left, int right) {
        
        if (left > right) return nullptr;
        int best = left;
        for (int i = left + 1; i <= right; i ++) if (nums[i] > nums[best]) best = i;

        TreeNode *node = new TreeNode(nums[best]);
        node->left = construct(nums, left, best - 1);
        node->right = construct(nums, best + 1, right);
        return node;
    }

    TreeNode *constructMaximumBinaryTree(vector<int> &nums) {
        return construct(nums, 0, nums.size() - 1);
    }
};

// 二叉搜索树中序遍历 1，2，3，4，5，6......
class Solution {
public:
    typedef long long ll;

    bool isValidBST(TreeNode *root) {
        stack<TreeNode *> stk;
        ll inorder = (ll) INT_MIN - 1;
        while (! stk.empty() || root) {
            while (root) {
                stk.push(root);
                root = root->left;
            }
            root = stk.top();
            stk.pop();
            if (root->val <= inorder) return false;
            inorder = root->val;
            root = root->right;
        }
        return true;
    }
};

 *@number[530]:二叉搜索树的最小绝对差
 *@author:lls86
 *@desc:dfs
 *@reference:
 *@time:2022-12-06 14:28:32
 */

class Solution {
public:
    void dfs(TreeNode *root, int &pre, int &ans) {
        if (root == nullptr) {
            return;
        }
        dfs(root->left, pre, ans);
        if (pre == - 1) {
            pre = root->val;
        } else {
            ans = min(ans, root->val - pre);
            pre = root->val;
        }
        dfs(root->right, pre, ans);
    }

    int getMinimumDifference(TreeNode *root) {
        int ans = INT_MAX, pre = - 1;
        dfs(root, pre, ans);
        return ans;
    }
};

 *@number[530]:二叉搜索树的最小绝对差
 *@author:lls86
 *@desc:层遍历，堆
 *@reference:
 *@time:2022-12-06 14:28:32
 */

class Solution {
public:
    int getMinimumDifference(TreeNode *root) {
        stack<TreeNode *> stk;
        TreeNode *cur = root;
        TreeNode *pre = nullptr;
        int result = INT_MAX;
        while (cur || ! stk.empty()) {
            if (cur) {
                stk.push(cur);
                cur = cur->left;
            } else {
                cur = stk.top();
                stk.pop();
                if (pre) {
                    result = min(result, cur->val - pre->val);
                }
                pre = cur;
                cur = cur->right;
            }
        }
        return result;
    }
};

 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-06 15:42:23

class Solution {
private:
    int maxCount = 0; // 最大频率
    int count = 0; // 统计频率
    TreeNode *pre = NULL;
    vector<int> result;

    void searchBST(TreeNode *cur) {
        if (cur == NULL) return;
        searchBST(cur->left);       // 左
        // 中
        if (pre == NULL) { // 第一个节点
            count = 1;
        } else if (pre->val == cur->val) { // 与前一个节点数值相同
            count ++;
        } else { // 与前一个节点数值不同
            count = 1;
        }
        pre = cur; // 更新上一个节点
        if (count == maxCount) { // 如果和最大值相同，放进result中
            result.push_back(cur->val);
        }
        if (count > maxCount) { // 如果计数大于最大值频率
            maxCount = count;   // 更新最大频率
            result.clear();     // 很关键的一步，不要忘记清空result，之前result里的元素都失效了
            result.push_back(cur->val);
        }
        searchBST(cur->right);      // 右
        return;
    }
public:
    vector<int> findMode(TreeNode *root) {
        count = 0;
        maxCount = 0;
        TreeNode *pre = NULL; // 记录前一个节点
        result.clear();

        searchBST(root);
        return result;
    }
};

        TreeNode *node = new TreeNode(val);
        if (! root) {
            return node;
        }
        TreeNode *cur = root;
        TreeNode *par = nullptr;
        while (cur) {
            if (cur->val > val) {

                cur = cur->left;

                par = cur;
                cur = cur->right;

            par->left = node;
        } else {
            par->right = node;
        }
        return root;
    }

/*
 *@number[450]:删除二叉搜索树中的节点
 *@author:lls86
 *@desc:dfs
 *@reference:
 *@time:2022-12-13 14:59:36
 */

class Solution {
public:
    TreeNode *deleteNode(TreeNode *root, int key) {
        if (! root) return root;  // 空
        if (root->val == key) {  // 当找到节点时
            if (root->left == nullptr && root->right == nullptr) {  // 左右节点为空
                delete root;
                return nullptr;
            } else if (root->left == nullptr) {  // 左节点空
                auto retnode = root->right;
                delete root;
                return retnode;
            } else if (root->right == nullptr) {  // 右节点空
                auto retnode = root->left;
                delete root;
                return retnode;
            } else {
                // 左右节点不空时，将删除节点的左子树根节点放到节点的右子树的最左面节点的左孩子的位置
                TreeNode *cur = root->right;
                while (cur->left != nullptr) {  // 找到节点右子树最左面的节点
                    cur = cur->left;
                }
                cur->left = root->left;  // 右子树最左面节点指向左子树根节点
                root = root->right;  // 要删除的节点变为右节点
                return root;
            }
        }
        if (root->val > key) root->left = deleteNode(root->left, key);
        if (root->val < key) root->right = deleteNode(root->right, key);
        return root;
    }
};

 *@desc:bfs

 *@time:2022-12-13 16:01:56

class Solution {
public:
    TreeNode *trimBST(TreeNode *root, int low, int high) {
        if (! root) return nullptr;
        // 处理头结点，让root移动到[low, high]范围内，注意是左闭右闭
        while (root != nullptr && (root->val < low || root->val > high)) {
            if (root->val < low) root = root->right;
            else root = root->left;
        }
        TreeNode *cur = root;
        // 此时root已经在[low, high]范围内，处理左孩子元素小于low的情况
        while (cur) {
            while (cur->left && cur->left->val < low) {
                cur->left = cur->left->right;
            }
            cur = cur->left;
        }
        cur = root;
        // 此时root已经在[low, high]范围内，处理右孩子元素大于high的情况
        while (cur) {
            while (cur->right && cur->right->val > high) {
                cur->right = cur->right->left;
            }
            cur = cur->right;
        }
        return root;
    }
};

        return result;

/*
 *@number[131]:分割回文串
 *@author:lls86
 *@desc:
 *@reference:
 *@time:2022-12-16 20:05:17
 */

class Solution {
private:
    vector<vector<string>> result;
    vector<string> path;

    bool isPalindrome(string &s, int start, int end) {  // 回文字符判断
        for (int i = start, j = end; i < j; i ++, j --) {
            if (s[i] != s[j]) return false;
        }
        return true;
    }

    void dfs(string &s, int startindex) {
        if (startindex == s.size()) {
            result.emplace_back(path);
            return;
        }

        // 先一个一个切割，当切割完之后，再回到顶层，startindex = 0， i = 1
        for (int i = startindex; i < s.size(); i ++) {
            if (isPalindrome(s, startindex, i)) {
                string str = s.substr(startindex, i - startindex + 1);  // substr[) 左闭右开
                path.emplace_back(str);
            } else {
                continue;
            }
            dfs(s, i + 1);
            path.pop_back();
        }
    }

public:
    vector<vector<string>> partition(string s) {
        dfs(s, 0);
        return result;
    }
};
